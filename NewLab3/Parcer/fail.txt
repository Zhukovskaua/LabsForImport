from html.parser import HTMLParser
 
is_price = False
class Parser(HTMLParser):
    def handle_starttag(self, tag, attrs):
        global is_price
        if tag == 'td':
            for attr in attrs:
                if attr[0] == 'class' and attr[1] == 'price':
                    is_price = True
                    break
    def handle_data(self, data):
        global is_price
        if is_price == True:
            print('Цена:', data)
            is_price = False
 
def main():
    f = open('test1.txt', 'r')
    html = f.read()
    parser = Parser()
    parser.feed(html)
    input()
 
if __name__ == '__main__':
    main()
